CREATE TABLE IF NOT EXISTS album(
  album_id SMALLINT(4) NOT NULL,
  album_title VARCHAR(128) NOT NULL,
  is_owned BOOLEAN NOT NULL,
  pc VARCHAR(128) NOT NULL,
  pc_path VARCHAR(128) NOT NULL,
  PRIMARY KEY (album_id)
)
SELECT album TABLE CREATED

CREATE TABLE IF NOT EXISTS version(
  album_id SMALLINT(4) NOT NULL,
  v_id SMALLINT(4) NOT NULL,
  version_name VARCHAR(128) NOT NULL,
  is_owned BOOLEAN NOT NULL,
  pc VARCHAR(128),
  pc_path VARCHAR(128),
  PRIMARY KEY (album_id, v_id)
)
